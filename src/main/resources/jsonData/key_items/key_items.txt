            "texture": String URI of inventory icon texture from images/sprites (autofills .png)
        },

                "effects": List of effects applied on use:[
                    {
                        "effect": String ID of effect to apply,
                        "duration": int duration of effect in seconds (omit for permanent until next world)
                    }
                ]
            }
        ]
    }
