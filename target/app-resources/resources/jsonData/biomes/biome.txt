{
    [String ID: snake_case String ID to identify biome]: {
        "name": String to represent biome's name in text,
        "level": Minimum level the biome can spawn at (default 1),
        "colorMap": {
            "voidColor": String HEX color of the biome's background color,
            "fogColor": String HEX color of the biome's fog (required fog tag),
            [NatColor String ID]: String HEX color of the biome's given natColor
        },
        "tags": List of following String tags of biome:[
            "fog", - Include for fog effect
            "watery", - Include to double generation of water down the center
        ]
        "generation": {
            "size": [int biome width, int biome height] (default [51, 100]),
            "start": [int start x pos, int start y pos] (default [biome width/2, biome height - 1])
            "startMap": 2D Array of biome's starting tile IDs (default [[1, 1, 1]]),
            "base": The int tile ID of the biome's base tile (default 0),
            "genSize": [int tiles generated from start, int tiles generated from biome] (default [biome width/2, biome height - 1]),
            "tiles": List of following Tile generation data:[
                {
                    "id": int ID of Tile to be generated
                    "prob": Number base probability of tile generating on any space,
                    "sideProb": [int Tile ID, Number probability added if Tile of given ID towards start],
                    "bottomProb": [int Tile ID, Number probability added if Tile of given ID below],
                    "rangeProb": [int minimum x pos of range, int maximum x pos of range, Number prbablity added if within given range]
                }
            ],
            "objects": List of following Object generation data:[
                {
                    "id": int ID of Object to be generated,
                    "prob": Number base probability of object generating on any space,
                    "tileProb": [int Tile ID, Number probability added if on Tile of given ID]
                }
            ],
            "entities": List of following Entity generation data:[
                {
                    "id": int ID of Entity to be generated,
                    "prob": Number base probability of entity to be generated on any space,
                    "tileProb": [int Tile ID, Number probability added if on Tile of given ID],
                    "levelProbPower": Number Power of scaling prbability from increasing level
                }
            ],
            "structures": List of following Structure generation data:[
                {
                    "id": int ID of Structure to be generated,
                    "pos": [int x pos of structure, int y pos of structure]
                }
            ]
        }
    }
}
