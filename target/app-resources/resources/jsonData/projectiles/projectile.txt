{
    [String ID: snake_case String ID to identify projectile]: {
        "id": int ID of projectile,
        "name": String to represent projectile's name in text,
        "texture": {
            "texture": String URI of texture file from images/sprites (autofils .png)
        },
        "tags": List of following String tags of projectile:[
            "hurtEntity", - Include to allow entities to be damaged by projectile
            "hurtPlayer", - Include to allow players to be damaged by projectile
            "hurtSource", - Include to allow the projectile's source to be damaged by projectile
            "alwaysRender", - Include to always render / update projectile
            "damageTile", - Include to allow objects to be damaged by projectile
        ],
        "attributes": {
            "damage": Number of damage to deal on contact (default 0.25),
            "size": Number of tiles of the projectile's size (default 0.5),
            "weight": Number of the projectile's weight (default 1),
            "speed": Number of tiles / second of the projectile's speed (default 0.25),
            "parryProj": int ID the projectile is replaced by on parry (default id)
        }
    }
}
